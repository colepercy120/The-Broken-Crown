	on_monthly_pulse = {
		random_events = {
			10 = 0
			1 = spanish_collapse.10
		}
		
	}
	
	modifiers_while_active = {
		bc_unrest
	}
	
	immediate = {
		every_scope_state = {
			add_radicals_in_state = {
				value = 1
			}
		}
		
		
	}
	
	
	

	# note that if you change this, also update the decision
	complete = {
		is_subject = no
		any_scope_state = {
			count >=10
			is_incorporated = yes
			is_split_state = no
			turmoil < 0.2
		}
		gdp >= 2000000
		total_population >= 2000000
	}

	on_complete = {
		
		
	}
	
	fail = {
		any_scope_state = {
			count <=3
			owner = root
		}
	}
	
	on_fail = {
		trigger_event = { id = spanish_collapse.11 popup = yes }
		
	}
	
	timeout = 3650

	on_timeout = {
		trigger_event = { id = spanish_collapse.11 popup = yes }
	}
	

	weight = 100

	transferable = no
	should_be_pinned_by_default = yes
}
