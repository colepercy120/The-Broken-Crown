je_alaska_settle_interior = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_objectives
	
	

	# note that if you change this, also update the decision
	complete = {
		calc_true_if = {
			amount > 3
			owns_entire_state_region = STATE_IDAHO
			owns_entire_state_region = STATE_WASHINGTON
			owns_entire_state_region = STATE_OREGON
			owns_entire_state_region = STATE_BRITISH_COLUMBIA
			owns_entire_state_region = STATE_YUKON_TERRITORY
			owns_entire_state_region = STATE_ALASKA
			}
		
		total_population > 500000
	}

	on_complete = {
		trigger_event = { id = alaska.10 }
		show_as_tooltip = {
			add_modifier = {
				name = alaska_settled
				years = 20 
 
				is_decaying = yes 
			}
			add_journal_entry = { type = je_alaska_national_identity }
		}
	}

	modifiers_while_active = {
		alaska_settlement
	}

	weight = 100

	transferable = no
	should_be_pinned_by_default = yes 
}

je_alaska_national_identity = {
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	group = je_group_objectives
	
	
	
	immediate = {
		add_primary_culture = cu:alaskan
		remove_primary_culture = cu:russian
		every_scope_pop = {
			limit = { culture = cu:russian } 
			change_pop_culture = {
				target = cu:alaskan
				value = 0.10
			}
		}
		every_scope_state = {
			state_region = {
				add_homeland = cu:alaskan
			}
		}
	}

	

	# note that if you change this, also update the decision
	complete = {
		prestige >= 50
		primary_cultures_percent_country >= 0.5
		total_population >= 1000000
	}

	on_complete = {
		trigger_event = { id = alaska.11 }
		show_as_tooltip = {
			add_modifier = {
				name = alaska_culture
				years = 20 
 
				is_decaying = yes 
			}
		}
	}

	modifiers_while_active = {
		
	}

	weight = 100

	transferable = no
	should_be_pinned_by_default = yes 
}
