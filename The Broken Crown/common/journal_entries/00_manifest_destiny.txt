je_manifest_destiny_main = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_usa_manifest_destiny
	
	immediate = {
		add_journal_entry = { type = je_manifest_destiny_mexico }
		add_journal_entry = { type = je_manifest_destiny_frontier_wars }
		add_journal_entry = { type = je_manifest_destiny_canada }
		add_journal_entry = { type = je_manifest_destiny_carribean }
		add_journal_entry = { type = je_manifest_destiny_pacific }
	}

	# note that if you change this, also update the decision
	complete = {
		calc_true_if = {
			amount > 5
			custom_tooltip = {
				text = manifest_destiny_mexico_tooltip
				has_global_variable = manifest_destiny_mexico
			}
			custom_tooltip = {
				text = manifest_destiny_frontier_tooltip
				has_global_variable = manifest_destiny_frontier
			}
			custom_tooltip = {
				text = manifest_destiny_canada_tooltip
				has_global_variable = manifest_destiny_canada
			}
			custom_tooltip = {
				text = manifest_destiny_carribean_tooltip
				has_global_variable = manifest_destiny_carribean
			}
			custom_tooltip = {
				text = manifest_destiny_pacific_tooltip
				has_global_variable = manifest_destiny_pacific
			}
		}
	}

	on_complete = {
		trigger_event = { id = manifest_destiny.7 }
	}

	weight = 100
	should_be_pinned_by_default = yes
}
