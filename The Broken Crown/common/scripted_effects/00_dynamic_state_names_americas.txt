STATE_OHIO_state_name_assign = {
    if = {
        limit = {
            owner = {
                c:UOM ?= this
            }
        }

        set_state_name = STATE_OHIO_michigander
    }
    if = {
        limit = {
            owner = {
                country_has_primary_culture = cu:japanese
            }
        }

        set_state_name = STATE_OHIO_japanese
    }
    else = { # switch back to defaults
        reset_state_name = yes
        reset_hub_names = yes
    }
}

STATE_WASHINGTON_state_name_assign = {
	if = {
		limit = {
			owner = {
				c:USA ?= this
			}
		}

        set_state_name = STATE_WASHINGTON_usa

        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_WASHINGTON_city_usa
        }
	}
	else_if = {
		limit = {
			owner = {
				any_primary_culture = {
						OR = {
							cu:alaskan = this
							cu:russian = this
						}
					}
			}
		}

        set_state_name = STATE_WASHINGTON_alaska

        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_WASHINGTON_city_alaska
        }
	}
	else = { # switch back to defaults
        reset_state_name = yes
        reset_hub_names = yes
    }
}

STATE_BRITISH_COLUMBIA_state_name_assign = { # More likely that British Columbia gets its name from conclusion of Oregon JE, but this serves as a one-off fallback
	if = {
		limit = {
			owner = {
				OR = {
					c:GBR ?= this
					is_subject_of = c:GBR
					any_primary_culture = {
						OR = {
							cu:anglo_canadian = this
							cu:franco_canadian = this
						}
					}
				}
			}
			s:STATE_WASHINGTON = {
				any_scope_state = {
					owner = {
						c:USA ?= this
					}
				}
			}
		}

        set_state_name = STATE_BRITISH_COLUMBIA_british
	}
	else_if = {
		limit = {
			owner = {
				any_primary_culture = {
						OR = {
							cu:alaskan = this
							cu:russian = this
						}
					}
			}
		}

        set_state_name = STATE_BRITISH_COLUMBIA_alaska

        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_BRITISH_COLUMBIA_city_alaska
        }
	}
	else_if = {
		limit = {
			owner = {
				c:USA ?= this
			}
		}

        set_state_name = STATE_BRITISH_COLUMBIA_usa

        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_BRITISH_COLUMBIA_city_usa
        }
	}
	else = { # switch back to defaults
        reset_state_name = yes
        reset_hub_names = yes
    }
}

STATE_ONTARIO_state_name_assign = {	# this effect is also covered by completing je_canada_can or je_canada_gbr but I am redundantly covering all bases in a redundant manner
	if = {
		limit = {
			owner = {
				OR = {
					country_has_primary_culture = cu:anglo_canadian
					country_has_primary_culture = cu:franco_canadian
				}
				has_technology_researched = nationalism
			}
		}

        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_ONTARIO_city_ottawa
        }
	}
	# no name reversion effect in an else statement. If the city hub becomes Ottawa, it stays Ottawa
}

STATE_DISTRICT_OF_COLUMBIA_state_name_assign = {
    if = {
        limit = {
            owner = {
                OR = {
                    c:GBR ?= this
                    AND = {
                        exists = top_overlord
                        exists = c:GBR
                        NOT = { c:USA ?= this }
                        top_overlord = c:GBR
                    }
                }
            }
        }

        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_DISTRICT_OF_COLUMBIA_city_uk
        }
    }
    else = { # switch back to defaults
        reset_state_name = yes
        reset_hub_names = yes
    }
}
